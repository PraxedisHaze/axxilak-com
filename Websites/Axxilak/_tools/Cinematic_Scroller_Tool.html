AXXILAK CINEMATIC SCROLLER v1.0

This tool automates smooth scrolling for perfect screen recordings.

Instructions:

  1. Open the webling you want to record.
  2. Press F12 (or Right Click > Inspect) to open the Developer Console.
  3. Copy the code below and paste it into the console.
  4. Press Enter.
  5. A control panel will appear. Click INITIATE to start scrolling.
  6. Press 'H' to hide the control panel before you start recording.
(function() { const UI_ID = 'axxilak-scroller-ui'; if (document.getElementById(UI_ID)) return; let config = { maxV: 1.5, accel: 0.01, brake: 0.02, buffer: 300, running: false, v: 0 }; const ui = document.createElement('div'); ui.id = UI_ID; ui.style.cssText = 'position:fixed;bottom:20px;right:20px;width:220px;background:rgba(0,0,0,0.95);color:#d4af37;padding:15px;border:1px solid #d4af37;border-radius:4px;font-family:monospace;z-index:9999999;box-shadow:0 0 20px rgba(212,175,55,0.2);display:flex;flex-direction:column;gap:10px;font-size:11px;'; ui.innerHTML = `
AXXILAK CINE-SCROLL
PRESS 'H' TO HIDE UI
`; document.body.appendChild(ui); const btn = document.getElementById('as-go'); function step() { if (!config.running) return; const max = document.documentElement.scrollHeight - window.innerHeight; const remain = max - window.pageYOffset; if (remain <= 0) { config.running = false; btn.innerText = "COMPLETE"; return; } if (remain < config.buffer) { config.v -= config.brake; if(config.v<0.2)config.v=0.2; } else if (config.v < config.maxV) { config.v += config.accel; } window.scrollBy(0, config.v); requestAnimationFrame(step); } btn.onclick = () => { config.running = !config.running; config.maxV = parseFloat(document.getElementById('as-max').value); if(config.running){btn.innerText="STOP";step();}else{btn.innerText="INITIATE";} }; window.addEventListener('keydown', e => { if(e.key.toLowerCase()==='h') ui.style.display = ui.style.display==='none'?'flex':'none'; }); })();
